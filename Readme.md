##### Apt

``` javascript
{
  apt: {
    run_at: 801,
    deb_options: [
      "mariadb-server-5.5 mysql-server/root_password password asdf",
    ],
    packages: [
      'mariadb-server-5.5',
    ],
    purge: [
      'mariadb-server-5.1',
    ],
  }
}
```
